How to use StartrJs.  Full Documentation.


1.   select(id)         // id - Any HTML tag id.

use :     select("t")       //   t   ->   <p id="t">Hello</p>.

2.   getVal(id)         // id - Any HTML tag id.

use :     getVal("name")       //   name   ->   <input type="text" id="name"></input>.

3.   setTxt(id,msg)       // id - Any HTML tag id.  | msg  ->  set your text here

use :     setTxt("span","Sample Text.")       //   span   ->   <span id="span">Sample Text.</span>.

4.   setHtml(id,html)       // id - Any HTML tag id.  | html  ->  code html here

use :     setHtml("t1", '<b>Sample Text.</b>')       //   t1   ->   <p id="t1">Sample Text.</p>.

5.   btnclk(id,event)       // id - Any HTML tag id.  | event  ->  code event function.

use :     btnclk("t",()=> {
   setTxt("span","Sample Text.")
})       //   t   ->   <button id="t"></button>.

6.   keyUp(id,event)       // id - Any HTML tag id.  | event  ->  code event function.

use :     keyUp("t",()=> {
   let name = getVal('name')
   alert(name)
})       //   t   ->   <input type="text" id="t"></input>.

7.   addClass(id,class)       // id - Any HTML tag id.  | class  ->  set css class

use :     addClass("t","phone")       //   t   ->   <p id="t" class="phone"></p>.

8.   removeClass(id,class)       // id - Any HTML tag id.  | class  ->  remove css class

use :     removeClass("t","phone")       //   t   ->   <p id="t"></p>.

9.   show(id)         // id - Any HTML tag id.

use :     show("t")       //   t   ->   <p id="t" style="display:block;"></p>.

10.   hide(id)         // id - Any HTML tag id.

use :     hide("t")       //   t   ->   <p id="t" style="display:none;"></p>.

11.   txtcolor(id,color)       // id - Any HTML tag id.  | color  ->  color name

use :     txtcolor("t","red")       //   t   ->   <p id="t">Sample Text.</p>.

12.   Ajax(url,method,data,respons)   // url - like.php, main.asp, dex.js  | method  ->POST / GET | data = { st: 1, name: "admin" }

use :
    POST   :-
Ajax({
    url:  "/api/checkUser",
    method: "POST",
    data: {
    st: true,
    userId: "uid_1011"
     },
    success: (respons) => {
     setTxt("status",respons)
   },
   error: (respons) => {
     console.log(respons)
   }
})
   GET :-
Ajax({
    url:  "/api/checkUser",
    method: "GET",
    success: (respons) => {
     setTxt("status",respons)
   },
   error: (respons) => {
     console.log(respons)
   }
})
